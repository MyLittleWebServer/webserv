139void Client::sendResponse() {
140 const std::string &response = _response.getResponse();
141 ssize_t n = send(_sd, response.c_str() + _lastSentPos,
142 response.size() - _lastSentPos, 0);
143 if (n == -1) throw Client::SendFailException();
144 if (n <= 0) throw Client::DisconnectedDuringSendException();
145
146 if (static_cast<size_t>(n) != response.size() - _lastSentPos) {
147 _lastSentPos += n;
148 return;
149 }
150 if (_request.getHeaderField("connection") == "close") {
151 _state = END_CLOSE;
152 return;
153 }
154 _state = END_KEEP_ALIVE;
155}
