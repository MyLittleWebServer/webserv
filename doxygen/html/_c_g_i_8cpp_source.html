114void CGI::generateResponse() {
115 _cgi_status = CGI_END;
116 size_t ret = _cgiResult.find("Status:");
117 if (ret != 0) {
118 generateErrorResponse(E_500_INTERNAL_SERVER_ERROR);
119 return;
120 }
121 ret = _cgiResult.find("Status: 200 OK");
122 if (ret != 0) {
123 generateErrorResponse(E_400_BAD_REQUEST);
124 return;
125 }
126 ssize_t start = _cgiResult.find("\r\n");
127 ret = _cgiResult.find("\r\n\r\n");
128 std::string header = _cgiResult.substr(start + 2, ret + 2);
129 std::string body = _cgiResult.substr(ret + 4);
130 _response->clear();
131 _response->setStatusCode(E_200_OK);
132 while (true) {
133 ret = header.find("\r\n");
134 if (ret == std::string::npos) break;
135 std::string line = header.substr(0, ret);
136 size_t colon = line.find(":");
137 if (colon == std::string::npos) break;
138 std::string key = line.substr(0, colon);
139 std::string value = line.substr(colon + 1);
140 _response->setHeaderField(key, value);
141 header = header.substr(ret + 2);
142 }
143 _response->setBody(body);
144 _response->setHeaderField("Content-Length", itos(body.length()));
145 _response->assembleResponse();
146 Kqueue::enableEvent(_client_fd, EVFILT_WRITE, _client_info);
147}
