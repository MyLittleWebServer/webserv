69 value = getWord(config_data, ind, ";");

71 value.substr(value.length() - f_len, f_len) == file_name)

100IServerConfig* ConfigParser::parseAndGenerateServer(
101 int& ind, const std::string& config_data) {
102 std::string key = getWord(config_data, ind, WHITE_SPACE);
103 std::string value;
104 keyCheck(key, "{", INVALID_SERVER_CONFIG);
105 IServerConfig* serverConfig = new ServerConfig();
106
107 while (true) {
108 try {
109 key = getWord(config_data, ind, WHITE_SPACE);
110 if (key == "}") return serverConfig;
111 if (key == "location") {
112 value = getWord(config_data, ind, WHITE_SPACE);
113 serverConfig->addLocationConfig(
114 parseAndGenerateLocation(ind, config_data, value));
115 continue;
116 }
117 value = getWord(config_data, ind, ";");
118 serverConfig->setVariable(key, value);
119 } catch (const ExceptionThrower::NposException& e) {
120 throw ExceptionThrower::InvalidConfigException(INVALID_SERVER_CONFIG);
121 }
122 }
123 return serverConfig;
124}

146std::string ConfigParser::getWord(const std::string& config_data, int& ind,
147 std::string delimeter) {
148 size_t start = config_data.find_first_not_of(WHITE_SPACE, ind);
149 nposCheck(start);
150 size_t end = config_data.find_first_of(delimeter, start);
151 nposCheck(end);
152 std::string result = config_data.substr(start, end - start);
153 ind = end + 1;
154 return result;
155}
