50void RequestParser::parseRequestLine(RequestDts &dts) {
51 const std::string &firstLine(dts.linesBuffer->front());
52 int delim_cnt = 0;
53 size_t pos = firstLine.find(" ", 0);
54
55 while (pos != std::string::npos) {
56 delim_cnt++;
57 pos = firstLine.find(" ", ++pos);
58 }
59 if (delim_cnt != 2) throw(*dts.statusCode = E_400_BAD_REQUEST);
60
61 std::istringstream iss(firstLine);
62 dts.linesBuffer->pop_front();
63 iss >> *dts.method >> *dts.path >> *dts.protocol;
64 size_t anchorPos = dts.path->find("#");
65 if (anchorPos != std::string::npos) parseAnchor(dts, anchorPos);
66 size_t qMarkPos = dts.path->find("?");
67 if (qMarkPos != std::string::npos) parseQueryString(dts, qMarkPos);
68
69 std::cout << "method: " << *dts.method << std::endl;
70 std::cout << "path: " << *dts.path << std::endl;
71 std::cout << "protocol: " << *dts.protocol << std::endl;
72 if (*dts.method == "" || *dts.path == "" || *dts.protocol == "")
73 throw(*dts.statusCode = E_400_BAD_REQUEST);
74 checkRequestUriLimitLength(dts);

124void RequestParser::parseHeaderFields(RequestDts &dts) {
125 std::list<std::string>::const_iterator lineIt = dts.linesBuffer->begin();
126 std::list<std::string>::const_iterator lineEnd = dts.linesBuffer->end();
127
128 std::string key;
129 std::string value;
130 size_t pos = 0;
131 size_t end = 0;
132
133 while (lineIt != lineEnd) {
134 pos = (*lineIt).find(":");
135 end = (*lineIt).find("\r\n");
136 if (pos == std::string::npos || end == std::string::npos)
137 throw(*dts.statusCode = E_400_BAD_REQUEST);
138 key = std::string(toLowerString((*lineIt).substr(0, pos)));
139 value =
140 std::string(toLowerString((*lineIt).substr(pos + 1, end - pos - 1)));
141 value = ft_trimOWS(value);
142 validateHeaderKey(key, dts);
143 removeNotAscii(key);
144 removeNotAscii(value);
145 if (_candidateFields.find(key) != _candidateFields.end()) {
146 validateDuplicateInvalidHeaders(key, dts);
147 (*dts.headerFields)[key] = value;
148 }
149 ++lineIt;
150 }
151 dts.linesBuffer->clear();
152}

251void RequestParser::parseChunkedEncoding(RequestDts &dts) {
252 std::string body = *dts.body;
253 dts.body->clear();
254 *dts.contentLength = 0;
255 size_t pos = 0;
256 size_t end = 0;
257 size_t chunkSize = 0;
258 std::string chunk;
259
260 while (pos != std::string::npos) {
261 end = body.find("\r\n", pos);
262 chunkSize = std::strtol(body.substr(pos, end - pos).c_str(), NULL, 16);
263 if (chunkSize == 0) {
264 *dts.isParsed = true;
265 return;
266 }
267 chunk = body.substr(end + 2, chunkSize);
268 *dts.body += chunk;
269 *dts.contentLength += chunkSize;
270 pos = end + 2 + chunkSize + 2;
271 }
